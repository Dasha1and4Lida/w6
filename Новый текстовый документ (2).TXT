#include <iostream>
#include <locale.h>

using namespace std;

int main() {
    // Настройка для русского языка
    setlocale(LC_ALL, "Russian");
    
    int number;
    int count = 0;    // Счетчик чисел
    int sum = 0;      // Сумма чисел
    
    cout << "[+] Введите число: ";
    cin >> number;
    
    // Продолжаем, пока не введен 0
    while (number != 0) {
        count = count + 1;    // Увеличиваем счетчик
        sum = sum + number;   // Добавляем к сумме
        
        // Запрашиваем следующее число
        cout << "[+] Введите число: ";
        cin >> number;
    }
    
    // Проверяем, были ли введены числа (кроме 0)
    if (count == 0) {
        cout << "\n[!] Вы не ввели ни одного числа!\n";
    } else {
        // Вычисляем среднее арифметическое
        double average = (double)sum / count;
        
        // Выводим результат
        cout << "\n[+] Среднее арифметическое: " << average << endl;
    }
    
    return 0;
}
