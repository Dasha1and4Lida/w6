#include <iostream>
#include <locale.h>

using namespace std;

int main() {
    setlocale(LC_ALL, "Russian");
    
    int number;
    int count = 0;
    int sum = 0; 
    
    cout << "[+] Введите число: ";
    cin >> number;
    
    
    while (number != 0) {
        count = count + 1;
        sum = sum + number; 
        
       
        cout << "[+] Введите число: ";
        cin >> number;
    }
    
  
    if (count == 0) {
        cout << "\n[!] Вы не ввели ни одного числа!\n";
    } else {
       
        double average = (double)sum / count;
        
       
        cout << "\n[+] Среднее арифметическое: " << average << endl;
    }
    
    return 0;

}
